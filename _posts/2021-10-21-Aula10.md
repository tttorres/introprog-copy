{% highlight perl linenos %}#exemplo01 - placas.pl
#!/usr/bin/perl

# Plano C
# cerca de 14 linhas

$ok = 0; # default

print "Placa\:\n";

$placa = <STDIN>;

$ch7 = chop ( $placa );
if ( $ch7 =~ m/[0123456789]/ ) { $ok++ }

$ch6 = chop ( $placa );
if ( $ch6 =~ m/[0123456789]/ ) { $ok++ }

# outros digitos
# outras letras

if ( $ok == 4 ) { print "Possui quatro digitos!" }

exit;
