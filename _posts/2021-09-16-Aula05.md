# funcao split()
# transformar string em array

# strings
$var_music = "Rain-Drops-On-Roses-And-Whiskers-On-Kittens";
$var_Mando = "This is the way";

# transformar strings em arrays
@music = split('-', $var_music);
@Mando  = split(' ', $var_Mando);

print "$music[5]\n";  
print "$Mando[3]\n";
